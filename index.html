{

    pTest->m_nNumber        = 2;

1.3、参数对象（拷贝构造）

当对象作为函数的参数时，会调用一个特殊的构造函数—拷贝构造函数。该函数只有一个参数，类型为对象的引用。

浅拷贝
如果在函数调用时传递参数对象，参数会进行复制，形参是实参的副本，相当于拷贝构造了一个全新的对象。如果没有定义拷贝构造函数，编译器会对原生对象与拷贝对象中的各种数据成员直接进行数据复制，成为默认拷贝构造函数，这种拷贝属于浅拷贝。

class CTest{
public:
    CTest(){
        m_nNumber = 1;
    }
    int m_nNumber;
};
int _tmain(int argc, _TCHAR* argv[])
{
    CTest test1;
    CTest test2(test1);
}
======================================================
    CTest test1;

012413C1  call        CTest::CTest (12411CCh)  
    CTest test2(test1);
012413C6  mov         eax,dword ptr [test1]          ;取出test1中的m_nNumber成员
012413C9  mov         dword ptr [test2],eax          ;赋值给test2的m_nNumber成员

当成员变量为指针类型，指向资源数据时，如果test1释放了该资源，当test2使用指向的资源时，程序就会出错。这种情况下，就需要将指向的资源也拷贝一份副本，这时就需要深拷贝了。

深拷贝

class CMyString{
public:
    CMyString(){
        //初始化指针为NULL
        m_pString = NULL; 
    }
    //拷贝构造函数（深拷贝）
    CMyString(CMyString& obj)
    {
        int nLen = strlen(obj.m_pString);
        this->m_pString = new char[nLen + sizeof(char)];
        strcpy(this->m_pString , obj.m_pString);
    }
    ~CMyString()
    {
        if (m_pString != NULL)
        {
            delete [] m_pString;
            m_pString = NULL;
        }
    }
    void SetString(char* pString)
    {
        if (m_pString != NULL)
        {
            delete [] m_pString;
            m_pString = NULL;
        }
        int nLen = strlen(pString);
        m_pString = new char[nLen + sizeof(char)];
        strcpy(m_pString,pString);
    }
public:
    char* m_pString;
};

void Show(CMyString MyString)
{
    printf(MyString.m_pString);
}

int _tmain(int argc, _TCHAR* argv[])
{
    CMyString MyString;
    MyString.SetString("HelloWorld");
    Show(MyString);
    return 0;
}

逆向基础第二十篇：类和对象 之 “this指针”

By John·Huang

发表于 2015-05-20

写C++应该会经常用到this指针，但是我们在类中并没有找到this的定义，许多程序员只知道使用this指针，但是不知道this从何而来。今天，我们来解开this指针的神秘的面纱！

首先，this是一个指针，在32位环境下占4字节大小，保存的数据应为内存地址。

1、结构体指针

我们先分析一段结构体：

struct A 
{
    int        m_int;
    short    m_short;
};
int _tmain(int argc, _TCHAR* argv[])
{
    struct A a    = {1,2};
    struct A *p = &a;
    printf("%p",&p->m_int);
    printf("%p",&p->m_short);
    getchar();
}

下面是对结构体的反汇编的分析：

通过上图可以看到，结构体指针的地址就是此结构的首地址。

那么类对象呢？我们在下一小节继续分析。

2、this指针

我们编写一个测试类：

class CTest{
public:
    int m_int;
public:
    void SetNumber(int n)
    {
        m_int = n;
    }
};

int _tmain(int argc, _TCHAR* argv[])
{
    CTest Test;
    Test.SetNumber(10);
    printf("%p \r\n", &Test);
    printf("%p \r\n", &Test.m_int);
    getchar();
}

输出结果与我们预期的相同，对象地址与第一个元素的地址相同（0033FAE8），如下图：

下面我们分析反汇编代码：

通过上图的分析，可以得知：

  • this指针的地址 = 对象的首地址 = 第一个成员变量的地址。
  • 调用成员函数时，一般采用 ecx保存this指针，在成员函数内部会可以使用这个值。

3、stdcall调用方式的成员函数

C++中调用成员函数默认使用的是thiscall，thiscall中默认使用ecx传递this指针变量。但是，在stdcall中，this的传递方式是不同的。

测试代码：

class CTest{
public:
    int m_int;
public:
    //调用方式为stdcall
    void _stdcall SetNumber(int n)
    {
        m_int = n;
    }
};
int _tmain(int argc, _TCHAR* argv[])
{
    CTest Test;
    Test.SetNumber(10);
    printf("%p \r\n", &Test);
    printf("%p \r\n", &Test.m_int);
    getchar();
}

反汇编代码：

当使用stdcall时，编译器会隐含的传递一个this指针作为第一个参数，相当于：

void Function(this, parameter...)
{
    //..do something..
}

实际上，大多数面向对象的编程语言，都是这么传递this指针的。

如果有过python编程经验的人，应该注意到在调用成员函数时候，是显示的传递this指针：

class people:
    #定义了一个属性
    name = 'jack'       
    #定义了一个方法，这里显示的传递self（this）指针
    def printName(self): 
        print self.name

这里不讲python，仅供参考，对比学习。

逆向基础第十九篇：内存对齐

By John·Huang

发表于 2015-05-19

结构体和类在为数据成员分配内存时，结构体中当前数据类型长度为M，制定的对齐值为N，那么实际对齐值 Q = min(M,N) 其成员的地址安排在Q的倍数上。

测试代码：

struct tagTest{
    short nShort;
    int nInt;
} test;
int _tmain(int argc, _TCHAR* argv[])
{
    test.nShort = 1;
    test.nInt    = 2;
    int nSize = sizeof(test);
    printf("%d",nSize);
    getchar();
}

部分反汇编结果：

    test.nShort = 1;
00C32D1E  mov         eax,1  
00C32D23  mov         word ptr [test (0C37138h)],ax      ;这里是short类型的赋值
    test.nInt    = 2;
00C32D29  mov         dword ptr [test+4 (0C3713Ch)],2    ;这里test+4进行了对齐
    int nSize = sizeof(test);
00C32D33  mov         dword ptr [nSize],8                ;sizeof 的结果为8

对nShort赋值之后，应在地址00C3713A（0C37138h+2）处分配nInt的空间。 但是int类型占4个字节，与指定的对齐（VC6中默认的对齐是8字节）比较后，但是00C3713A不是4的倍数，需要插入两个字节补充，因此nInt被定义在了0C3713Ch处。

内存对齐实战：

第一：

struct{
    double nDouble;     //占8字节
    int nInt;           //占4字节
    short nShort;       //占2字节
}
按默认的对齐值要求，结构体的整体大小要能被8整除（8=结构的第一个元素长度），所以要 8+4+4+2+2=16

第二：

struct{
    char nChar;        //占4字节，与nInt对齐
    int nInt;          //占4字节
    short nShort;       //占2字节
}

第三：

#pragma pack(1)        //声明1字节对齐
struct{
    char nChar;        //占1字节
    int nInt;          //占4字节
    short nShort;       //占2字节
}

第四：

struct{
    char nChar;        //占1字节
    char cArray[4];    //占5字节，对齐nShort
    short nShort;       //占2字节
}

逆向基础第十七篇：数组和指针

By John·Huang

发表于 2014-05-17

数组和指针都是针对地址进行操作，但是他们有很多不同之处。数组是相同类型的数据集合，是以线性的方式存储在内存中；而指针是保存4个字节的地址变量。

数组名是一个地址常量，保存数组第一个元素的地址（不可以更改），只能作为基址访问内存数据；
指针是一个变量，只要修改所保存的地址数据，就可以随意访问，不受约束。

1、数组的定义

int _tmain(int argc, _TCHAR* argv[])
{
    int aArray[5] = {1,2,3,4,5};
    int var1 = 1;
    int var2 = 2;
    int var3 = 3;
    int var4 = 4;
    int var5 = 5;
}

2、数组作为参数

数组所占的内存，通常大于4个字节，那么它是如何传递参数的呢？
测试代码如下：

void Show(char szBuff[])
{
    strcpy(szBuff,"hello world");
    printf(szBuff);
}

int _tmain(int argc, _TCHAR* argv[])
{
    char szHello[20] = {0};
    Show(szHello);
}

反汇编代码：

通过上图，可以看出，当数组作为参数时，传递的是数组的首地址（指针变量）。因为是参数是指针变量，因此，使用sizeof只能得到指针长度，因此，应该避免如下错误代码：

void Show(char szBuff[])
{
    int nLen = 0;

    //这里是错误代码，得到的是指针长度,4字节
    nLen =  sizeof(szBuff);
    printf("%d ,",nLen);

    //正确的使用方法
    nLen = strlen(szBuff);
    printf("%d ,",nLen);

    getchar();
}

3、数组作为返回值

数组作为返回值时，与作为参数时相同，都是返回数组第一个元素的地址。注意，数组作为参数时，其定义的作用域必然在函数外，但是作为返回值时，如果数组在函数内定义，作为函数的局部变量，那么开辟的栈空间，在函数结束的时候要进行堆栈平衡，esp归位。这样，函数所在的内存空间就会极不稳定，随时可能会被破坏，影响程序的结果。
因此，如果想使用数组作为返回值，可以使用“全局数组，静态数组，或者上层调用函数中定义的局部数组”

4、下标寻址和指针寻址

逆向基础第十五篇：函数调用的三种约定方式（堆栈平衡）

By John·Huang

发表于 2014-05-15

VC++ 环境下的的函数调用约定有三种：

  • _cedcl C/C++的默认调用方式，调用方平衡栈，不定长参数的函数可以使用。
  • _stdcall 被调用方平衡栈，不定长参数的函数无法使用
  • _fastcall 寄存器方式传参，被调用方平衡栈，不定长参数的函数无法使用。

1、_cdecl 与 _stdcall的对比

void _stdcall ShowStd(int nNumber)
{
    printf("%d",nNumber);
}

void _cdecl ShowCde(int nNumber)
{
    printf("%d",nNumber);
}

int _tmain(int argc, _TCHAR* argv[])
{
    ShowStd(5);//被调用方平衡栈
    ShowCde(5);//调用方平衡栈
}

main 函数反汇编代码如下：

    ShowStd(5);
00CC1A6E  push        5  
00CC1A70  call        ShowStd (0CC11C7h)  
    ShowCde(5);
00CC1A75  push        5  
00CC1A77  call        ShowCde (0CC101Eh)  
00CC1A7C  add         esp,4  //这里对栈进行了平衡

_cdecl 与 _stdcall 函数的反汇编代码对比：

经过对比，我们发现只有最后一行有区别。

_stdcall 反汇编最后ret 4 等价于：esp += 4 对栈顶加4，用于平衡栈顶。
_cdecl被调用后，在main函数中对esp进行了加4操作，用于平衡栈顶。

2、_fastcall 调用方式

测试代码：

{

00AB141C  call        ShowFast (0AB11CCh)

由于寄存器的数量有限，而参数的数量有可能很多，因此fastcall仅使用了 edx和ecx，分别传递了第一个和第二个参数，其余的使用了栈传递参数。

逆向基础第十四篇：认识栈帧（堆栈平衡）

By John·Huang

发表于 2014-05-14

当栈顶指针esp小于栈底指针ebp时，就形成了栈帧。在VC++中，栈帧用来存放局部变量、函数返回值地址、函数参数等。

当A函数调用B函数时，就会针对B函数开辟出所需的栈空间，形成此B函数的栈帧。当这个B函数结束调用时，要清除开辟出的栈空间，这一过程，称作 “堆栈平衡”。

下面是一个空函数：

int _tmain(int argc, _TCHAR* argv[])
{
    return 0;
}

反汇编代码如下：

int _tmain(int argc, _TCHAR* argv[])
{
00F01350  push        ebp                  ;保存寄存器ebp的值
00F01351  mov         ebp,esp              ;调整当前栈底指针到栈顶（将ebp作为栈底指针）
00F01353  sub         esp,0C0h             ;分配192字节的空间
00F01359  push        ebx                  ;保存寄存器ebx的值
00F0135A  push        esi                  ;保存寄存器esi的值
00F0135B  push        edi                  ;保存寄存器edi的值
00F0135C  lea         edi,[ebp-0C0h]       ;取栈顶地址，保存在edi中
00F01362  mov         ecx,30h  
00F01367  mov         eax,0CCCCCCCCh  
00F0136C  rep stos    dword ptr es:[edi]  
    return 0;
00F0136E  xor         eax,eax              ;对返回值eax清零
}
00F01370  pop         edi                  ;还原寄存器edi的值
00F01371  pop         esi                  ;还原寄存器esi的值
00F01372  pop         ebx                  ;还原寄存器ebx的值
00F01373  mov         esp,ebp              ;还原寄存器esp的值
00F01375  pop         ebp                  ;还原寄存器ebp的值
00F01376  ret
  • rep指令的目的是重复其上面的指令。
  • ecx的值是重复的次数。
  • stos指令的作用是，将eax中的值拷贝到es:[edi]指向的地址。

逆向基础第十三篇：编译器对循环的优化

By John·Huang

发表于 2014-05-13

之前，我已经对do、while、和for循环做了详细的介绍。

do循环：http://www.0x520.com/2014/07/04/247.html
while循环：http://www.0x520.com/2014/07/04/251.html
for循环：http://www.0x520.com/2014/07/05/264.html

通过对比，我们已经知道，do循环的速度最快，while次之，for的速度是最慢的。 因为，do只用到一个跳转指令，while用到两个跳转指令，而for用到了三个跳转指令。

1、优化while循环

在 while循环 中我们讲过，是可以将while转换为do循环的，可以将do循环结构嵌套在if中，来完成while结构的优化。
while循环代码：

void loop_while(int loopTimes)
{
    while(loopTimes>=0)
    {
        //do something
        loopTimes --;
    }
}

优化代码：

void loop_do(int loopTimes)
{
    if(loopTimes>=0)
    {
        do{
            //do something
            loopTimes --;
        }while(loopTimes>=0);
    }
}

如果编译器使用O2选项进行优化，那么，反汇编之后的代码将是上面这种形式。

2、优化for循环

for循环，从结构上看，其结构特征和while循环类似。由于赋初值部分不属于循环体，可以忽略。只要将比较部分放入循环体，那么就是一个while循环。既然可以转换成while循环，那么必然也可以转换成do循环。

for循环代码：

int nSum = 0;
for (int i = 0;i < loop_times ; i++)
{
    nSum += i;
}

for转换成while循环：

int i = 0;
while(i<loop_times){
    nSum += i;
    i++;
}

while循环转换成do循环：

int nSum = 0;
int i = 0;
if(i<loop_times){
    do {
        nSum += i;
        i++;
    } while (i<argc);
}

编译器真的会这样优化吗？我们打开O2选项（选择Release，默认就是O2）：

编译如下代码：

int loop_for(int loop_times)
{
    int nSum = 0;
    for (int i = 0;i < loop_times ; i++)
    {
        nSum += i;
    }
    return nSum;
}

上图中for循环优化之后的结构一目了然，但是有些朋友表示还是看不懂，表示对下面这段代码不知所云：

add     edx,eax

这里我来讲解一下，其实这已经不是编译器对for循环优化的范畴了，这里是编译器对当前算法的优化 nSum += (i++);

  • edx 等于变量 nSum 的偶数和
  • ecx 等于变量 nSum 的奇数和
  • eax 等于变量 i

最后，在地址00401034处，lea eax,[ecx+edx]，这里 eax = 奇数和+偶数和 。
